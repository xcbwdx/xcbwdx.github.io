### 罗马数字

```c++
#include <stdio.h>
```
int main(){
	char a[15];
	int i=0,sum=0;
	scanf("%s",a);
	for(i=0;i<15;i++){
		if(a[i]=='I'&&a[i+1]=='V'){
			sum+=-1;
		}else if(a[i]=='I'&&a[i+1]=='X'){
			sum+=-1;
		}else if(a[i]=='X'&&a[i+1]=='L'){
			sum+=-10;
		}else if(a[i]=='X'&&a[i+1]=='C'){
			sum+=-10;
		}else if(a[i]=='C'&&a[i+1]=='D'){
			sum+=-100;
		}else if(a[i]=='C'&&a[i+1]=='M'){
			sum+=-100;
		}else if(a[i]=='I'){
			sum+=1;
		}else if(a[i]=='V'){
			sum+=5;
		}else if(a[i]=='X'){
			sum+=10;
		}else if(a[i]=='L'){
			sum+=50;
		}else if(a[i]=='C'){
			sum+=100;
		}else if(a[i]=='D'){
			sum+=500;
		}else if(a[i]=='M'){
			sum+=1000;
		}	
	}
	printf("%d",sum);	
	return 0;
}
